@Messages("title.install")

  1. At your dependencies add

    "pbaris" %% "play2-plugins-thumbs" % "1.0.1"

  2. At your resolvers add

    resolvers += "pbaris" at "https://github.com/pbaris/m2-repo/raw/master"

  3. Create a file called play.plugins in your application's conf directory and add the line 10000:pbaris.play.thumbs.ThumbsPlugin

  4. At your conf/routes add

    GET		/thumb		pbaris.play.thumbs.Thumbnail.thumb()
}
